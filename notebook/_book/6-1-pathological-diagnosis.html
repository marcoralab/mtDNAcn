tab_rosmap_abc <- rosmap.raw %>% 
  select(braaksc, ceradsc, niareagansc) %>% 
  count(ceradsc, braaksc) %>%
  mutate(ceradsc = as.character(ceradsc), 
         braaksc = as.character(braaksc), 
         abc = glue::glue("C{ceradsc}_B{braaksc}"))
