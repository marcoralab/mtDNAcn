  polr(niareagansc ~ apoe4 + sex + aod , data = ., Hess = TRUE)

tidy(model_fit) %>% 
  mutate(pval = pnorm(abs(statistic),lower.tail = FALSE)* 2) %>% 
  left_join(
   cbind(OR = exp(coef(model_fit)), 
          confint(model_fit)) %>% 
  as.data.frame() %>%
  rownames_to_column(), 
  by = c("term" = "rowname")
  ) %>% 
  clean_names()

sf <- function(y) {
  c('Y>=1' = qlogis(mean(y >= 1)),
    'Y>=2' = qlogis(mean(y >= 2)),
    'Y>=3' = qlogis(mean(y >= 3)))
}
