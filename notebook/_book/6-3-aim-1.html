gt.vglm <- function(model, tidy){
  # Pull out levels of model
  mod_levels = model@xlevels %>%
    map(., ~tibble(levels = .x)) %>%
    bind_rows(.id = "term") %>%
    mutate(mod = paste0(term, levels))
  
  # Sytle results
  out <- tidy %>% 
    filter(!str_detect(term, "Intercept")) %>%
    mutate(OR = round(OR, 2), 
           lci = round(lci, 2), 
           uci = round(uci, 2),
           `p-value` = round(p.value, 2),
           `95% CI` = paste0(lci, ', ', uci)) %>%
     select(var = term, OR, `95% CI`, `p-value`) %>%
    full_join(mod_levels, by = c("var" = "mod")) %>%
    mutate(term = ifelse(is.na(term), var, term),
           levels = ifelse(is.na(levels), var, levels),
           term = fct_inorder(term)) %>%
    replace_na(., list(OR = "—", `95% CI` = "—", `p-value` = "—")) %>%
    arrange(term, levels) %>%
    select(term, levels, OR, `95% CI`, `p-value`)
  out
  # return(out)
}
